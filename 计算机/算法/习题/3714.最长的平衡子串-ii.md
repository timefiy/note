## 解答

```cpp
class Solution {
public:
    int longestBalanced(string s) {
        int n = s.size();
        int ans = 0;

        for(int i = 0; i < n; i++){
            int cnt[26] = {0}; 
            
            for (int j = i; j < n; j++)
            {
                cnt[s[j] - 'a']++;

                if(check(cnt)){
                    ans = max(ans, j - i + 1);
                }
            }
        } 
        return ans;
    }

    bool check(int cnt[]){
        
    }
};
```
